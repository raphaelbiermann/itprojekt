44 void I2C_Master_Steady()
45 {
46  switch ( nI2CState )
47  {
48  case I2C_READY:
49  // wait for a new job
50  break;
51  case I2C_BUSY:
52  if ( Wire.available() )
53  {
54  int nReceived = 0;
55  char ch;
56  while ( Wire.available() )
57  {
58  ch = Wire.read(); // receive byte, typically I2C_DATA_MAX as requested
59  if ( nReceived < I2C_DATA_MAX )
60  szResponse[nReceived++] = ch; // store received byte
61  // else forget it
62  }
63  szResponse[nReceived] = 0; // make sure there is a trailing 0
64  nResponseLength = nReceived;
65  nResponseTime = micros() - nRequestTime;
66  nI2CState = I2C_DONE; // received complete, expect user fetches response
67  }
68  else if ( ( micros() - nRequestTime ) >= nResponseTimeOut )
69  { // timeout
70  nResponseTime = micros() - nRequestTime; //?? required
71  nI2CState = I2C_TIMEOUT;
72  }
73  break;
74  case I2C_DONE:
75  // remain here until response has been delivered
76  break;
77  case I2C_TIMEOUT:
78  // remain here until response has been asked
79  break;
80  }
81  delay(1);
82 }
